## 用法

.使用 Docker
[source,sh]
----
mkdir -p storage

docker run -it --rm \
  --name comfy3d-pt23 \
  --gpus all \
  -p 8188:8188 \
  -v "$(pwd)"/storage:/root \
  -e CLI_ARGS="" \
  yanwk/comfyui-boot:comfy3d-pt23
----

.使用 Podman
[source,sh]
----
mkdir -p storage

podman run -it --rm \
  --name comfy3d-pt23 \
  --device nvidia.com/gpu=all \
  --security-opt label=disable \
  -p 8188:8188 \
  -v "$(pwd)"/storage:/root \
  -e CLI_ARGS="" \
  docker.io/yanwk/comfyui-boot:comfy3d-pt23
----
